Send Bulk Requests

                
                
            

            

import httpx
import asyncio

async def send_bulk_requests():
async with httpx.AsyncClient() as client:
    tasks = [client.get("http://127.0.0.1:8000/async-task") for _ in range(20)]
    responses = await asyncio.gather(*tasks)
    for response in responses:
        print(response.json())
